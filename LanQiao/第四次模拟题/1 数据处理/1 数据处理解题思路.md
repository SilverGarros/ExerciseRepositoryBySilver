#### 解题思路

依次读取三类文件中的数据处理为要求python字典dict格式，合并三个字典数据。 `pandas`库处理一月数据xlsx； 二月json文件不需要大的处理； `Beautifulsoup`库解析处理HTML文件。

```python
import pandas as pd
import json
from bs4 import BeautifulSoup

# TODO 读取并处理xlsx文件
# 读取xlsx文件

data_1_xlsx = pd.read_excel('/home/project/2022_january.xlsx', skiprows=4,index_col=2)
# 删除前两列
data_1_xlsx = data_1_xlsx.drop(data_1_xlsx.columns[[0,1]],axis=1)
# 将日期列转换为日期格式，并设置为索引
data_1_xlsx.index = pd.to_datetime(data_1_xlsx.index)
# 转换格式
data_1_xlsx.index = data_1_xlsx.index.strftime('%-m-%-d') # 使用'-m''-d'去掉前导零
data_1 ={'january': data_1_xlsx.to_dict('index')}
# print(data_2)

# TODO 读取json文件(无需处理)
# 打开并读取json文件
with open('/home/project/2022_february.json','r')as jsonf:
    data_2 =json.load(jsonf)
# print(data_2)

# TODO 读取并解析HTML文件(使用BeautifalSoup库)
# 读取html文件
with open('/home/project/2022_may.html','r')as htmlf:
    data_5_html = htmlf.read()
data_5_html = BeautifulSoup(data_5_html,'lxml')

data_5_html_body = [body.text for body in data_5_html.find_all('th')]
print(data_5_html_body)
keys = data_5_html_body[:7]
print(keys)
data_5 = {"may":{}}
for i in range(7,len(data_5_html_body),7):
    date = data_5_html_body[i]
    print(date)
    values = data_5_html_body[i:i+7]
    print(values)
    date_dicts = None
    for j in range(1,7):
        date_dict ={keys[j]:int(values[j])}
        print("   "+str(date_dict))
        if date_dicts is not None:
            date_dicts = {**date_dicts,**date_dict}
        else :
            date_dicts = date_dict
    data_5['may'][date] = date_dicts

print(data_5)
    
data = {**data_1,**data_2,**data_5}

print(data)
```
